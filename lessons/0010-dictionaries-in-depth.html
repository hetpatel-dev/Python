Lesson 10: Dictionaries in Depth — Python's Mapping Type

concept Dictionaries

Dictionaries (dict) are Python's built-in mapping type — they store key-value pairs with O(1) average access time. Python's dict is one of the most versatile and optimized data structures in the language. This lesson covers everything from the official Python docs on dict.

Primary sources: Python Docs — Mapping Types, Data Structures Tutorial, collections module.

1. Dict Basics

Dictionaries are built with {} literal syntax or the dict() constructor. Keys must be hashable (immutable types like str, int, tuple). As of Python 3.7+, dicts preserve insertion order:

# Literal syntax
  d = {"name": "Alice", "age": 30, "city": "London"}
  print(d)        # {'name': 'Alice', 'age': 30, 'city': 'London'}

  # dict() constructor — from keyword arguments
  d2 = dict(name="Bob", age=25, city="Paris")
  print(d2)       # {'name': 'Bob', 'age': 25, 'city': 'Paris'}

  # dict() from a sequence of key-value pairs
  pairs = [("a", 1), ("b", 2), ("c", 3)]
  d3 = dict(pairs)
  print(d3)       # {'a': 1, 'b': 2, 'c': 3}

  # dict() from zip()
  keys = ["x", "y", "z"]
  vals = [10, 20, 30]
  d4 = dict(zip(keys, vals))
  print(d4)       # {'x': 10, 'y': 20, 'z': 30}

  # Dict comprehension
  squares = {x: x**2 for x in range(1, 6)}
  print(squares)  # {1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

  # Keys must be hashable
  # d_bad = {[1, 2]: "value"}  # TypeError: unhashable type: 'list'
  d_good = {(1, 2): "value"}   # tuple is hashable
  print(d_good)   # {(1, 2): 'value'}

  # Insertion order is preserved (3.7+)
  ordered = {"first": 1, "second": 2, "third": 3}
  print(list(ordered.items()))  # [('first', 1), ('second', 2), ('third', 3)]

  # Equality ignores order
  d5 = {"a": 1, "b": 2}
  d6 = {"b": 2, "a": 1}
  print(d5 == d6)  # True

  # Dicts cannot be ordered with < or > (raises TypeError)
  # print(d5 < d6)  # TypeError: '<' not supported between instances of 'dict' and 'dict'

2. Basic Operations

Standard dict operations: access, assign, delete, membership, and iteration:

d = {"name": "Alice", "age": 30, "city": "London"}

  # Access — raises KeyError if missing
  print(d["name"])          # Alice
  # print(d["country"])     # KeyError: 'country'

  # Assign (insert or update)
  d["country"] = "UK"       # new key
  d["age"] = 31             # update existing key
  print(d)                  # {'name': 'Alice', 'age': 31, 'city': 'London', 'country': 'UK'}

  # Delete
  del d["country"]
  print(d)                  # {'name': 'Alice', 'age': 31, 'city': 'London'}

  # Membership — check if key exists (O(1))
  print("name" in d)        # True
  print("country" in d)     # False

  # Length
  print(len(d))             # 3

  # Convert keys to a list
  print(list(d))            # ['name', 'age', 'city']

  # Iteration (iterates over keys by default)
  for k in d:
      print(k, d[k])        # name Alice  /  age 31  /  city London

  # reversed() — Python 3.8+
  print(list(reversed(d)))  # ['city', 'age', 'name']

  # Truthiness — empty dict is falsy
  print(bool({}))           # False
  print(bool({"key": 1}))   # True

3. The get() Method

Unlike direct bracket access, .get() never raises KeyError. It returns None (or a default of your choice) for missing keys:

d = {"name": "Alice", "age": 30}

  # Safe access with get()
  print(d.get("name"))           # Alice
  print(d.get("country"))        # None
  print(d.get("country", "N/A")) # N/A — custom default

  # Contrast with direct access
  # d["country"]  # KeyError!

  # Safe access pattern — avoids branching
  name = d.get("name", "Unknown")
  country = d.get("country", "Unknown")
  print(f"{name} from {country}")  # Alice from Unknown

  # get() does NOT insert the default into the dict
  print(d)                         # {'name': 'Alice', 'age': 30} — unchanged

  # See also: setdefault() inserts the default if missing

4. Dict Methods

Beyond get(), dicts offer a rich set of methods for manipulation:

d = {"a": 1, "b": 2, "c": 3}

  # pop(key[, default]) — remove and return value
  val = d.pop("a")
  print(val)      # 1
  print(d)        # {'b': 2, 'c': 3}

  val2 = d.pop("x", "not found")
  print(val2)     # 'not found' — no KeyError with default

  # d.pop("x")    # KeyError: 'x' — without default

  # popitem() — remove and return (key, value) in LIFO order (3.7+)
  d2 = {"a": 1, "b": 2, "c": 3}
  print(d2.popitem())  # ('c', 3) — last inserted
  print(d2.popitem())  # ('b', 2)
  print(d2.popitem())  # ('a', 1)
  # d2.popitem()       # KeyError: 'popitem(): dictionary is empty'

  # setdefault(key[, default]) — insert if missing, return value either way
  d3 = {"x": 10}
  print(d3.setdefault("x", 99))   # 10 — already exists, returns 10
  print(d3.setdefault("y", 99))   # 99 — inserted, returns 99
  print(d3)                       # {'x': 10, 'y': 99}

  # update([other]) — merge in key-value pairs
  d4 = {"a": 1, "b": 2}
  d4.update({"b": 99, "c": 3})
  print(d4)                       # {'a': 1, 'b': 99, 'c': 3}

  # update can take keyword args, an iterable of pairs, or zipped keys
  d4.update(d=4, e=5)
  print(d4)                       # {'a': 1, 'b': 99, 'c': 3, 'd': 4, 'e': 5}

  # clear() — remove all items
  d5 = {"a": 1, "b": 2}
  d5.clear()
  print(d5)                       # {}

  # copy() — shallow copy
  d6 = {"a": [1, 2]}
  d7 = d6.copy()
  d7["a"].append(3)
  print(d6)                       # {'a': [1, 2, 3]} — shared reference!
  print(d7)                       # {'a': [1, 2, 3]}

  # fromkeys(iterable, value) — class method, creates new dict
  keys = ["a", "b", "c"]
  d8 = dict.fromkeys(keys, 0)
  print(d8)                       # {'a': 0, 'b': 0, 'c': 0}

  # Mutable default trap — all values share the SAME list
  d9 = dict.fromkeys(keys, [])
  d9["a"].append(99)
  print(d9)                       # {'a': [99], 'b': [99], 'c': [99]}

  # Fix: use a dict comprehension
  d10 = {k: [] for k in keys}
  d10["a"].append(99)
  print(d10)                      # {'a': [99], 'b': [], 'c': []}
⚠️ popitem() order: In Python 3.6 and earlier, popitem() removed an arbitrary key-value pair. As of Python 3.7, it removes the most recently inserted item (LIFO). This is guaranteed behavior.

5. Dict Views

.keys(), .values(), and .items() return dynamic view objects — they reflect changes to the dict in real time:

d = {"a": 1, "b": 2, "c": 3}

  keys = d.keys()
  values = d.values()
  items = d.items()

  print(keys)      # dict_keys(['a', 'b', 'c'])
  print(values)    # dict_values([1, 2, 3])
  print(items)     # dict_items([('a', 1), ('b', 2), ('c', 3)])

  # Views are dynamic — they update when the dict changes
  d["d"] = 4
  print(keys)      # dict_keys(['a', 'b', 'c', 'd']) — automatically updated!
  print(items)     # dict_items([('a', 1), ('b', 2), ('c', 3), ('d', 4)])

  # Views support iteration (insertion order)
  for k in keys:
      print(k)     # a  b  c  d

  for k, v in items:
      print(k, v)  # a 1  /  b 2  /  c 3  /  d 4

  # Set-like operations on keys() — union, intersection, difference
  d1 = {"a": 1, "b": 2, "c": 3}
  d2 = {"b": 4, "c": 5, "d": 6}

  print(d1.keys() & d2.keys())   # {'b', 'c'} — intersection
  print(d1.keys() | d2.keys())   # {'a', 'b', 'c', 'd'} — union
  print(d1.keys() - d2.keys())   # {'a'} — difference
  print(d1.keys() ^ d2.keys())   # {'a', 'd'} — symmetric difference

  # .mapping attribute (3.10+) — get the underlying dict from a view
  print(keys.mapping is d)       # True

  # values() have set-like equality caveat
  print(d1.values() == d2.values())  # False — always False (by design)

  # Convert views to lists when you need a static copy
  keys_list = list(d1.keys())
  print(keys_list)               # ['a', 'b', 'c']
⚠️ Mutation during iteration: Modifying a dict while iterating over its views raises RuntimeError: dictionary changed size during iteration. Iterate over a copy of the keys when modifying: for k in list(d): or for k in d.copy():.

6. Iterating Dictionaries

Python provides several idiomatic patterns for dict iteration:

d = {"name": "Alice", "age": 30, "city": "London"}

  # Iterating over keys (default)
  for k in d:
      print(k)        # name  age  city

  # Iterating over key-value pairs
  for k, v in d.items():
      print(f"{k} = {v}")   # name = Alice  /  age = 30  /  city = London

  # Iterating over values
  for v in d.values():
      print(v)        # Alice  30  London

  # Sorted iteration
  for k in sorted(d):
      print(k, d[k])  # age 30  /  city London  /  name Alice

  # Reversed iteration (Python 3.8+)
  for k in reversed(d):
      print(k, d[k])  # city London  /  age 30  /  name Alice

  # enumerate over items
  for i, (k, v) in enumerate(d.items(), start=1):
      print(f"{i}. {k} = {v}")   # 1. name = Alice  /  2. age = 30  /  3. city = London

  # Unpacking items tuple explicitly
  for item in d.items():
      k, v = item
      print(k, v)

  # Modifying during iteration — always use list() wrapper!
  d2 = {"a": 1, "b": 2, "c": 3, "d": 4}
  # Remove items with value < 3
  for k in list(d2):      # iterate over a COPY of the keys
      if d2[k] < 3:
          del d2[k]
  print(d2)               # {'c': 3, 'd': 4}

7. Dict Comprehensions

Dict comprehensions build dicts concisely, with full support for filtering and transformation:

# Basic dict comprehension
  squares = {x: x**2 for x in range(1, 11)}
  print(squares)  # {1: 1, 2: 4, 3: 9, 4: 16, 5: 25, 6: 36, 7: 49, 8: 64, 9: 81, 10: 100}

  # With condition filter
  even_squares = {x: x**2 for x in range(1, 11) if x % 2 == 0}
  print(even_squares)  # {2: 4, 4: 16, 6: 36, 8: 64, 10: 100}

  # Building from two lists with zip()
  names = ["Alice", "Bob", "Charlie"]
  scores = [95, 87, 92]
  gradebook = {name: score for name, score in zip(names, scores)}
  print(gradebook)  # {'Alice': 95, 'Bob': 87, 'Charlie': 92}

  # Swapping keys and values
  original = {"a": 1, "b": 2, "c": 3}
  swapped = {v: k for k, v in original.items()}
  print(swapped)  # {1: 'a', 2: 'b', 3: 'c'}

  # Transforming values
  upper = {k: k.upper() for k in ["alice", "bob", "charlie"]}
  print(upper)  # {'alice': 'ALICE', 'bob': 'BOB', 'charlie': 'CHARLIE'}

  # Filtering a dict
  temperatures = {"London": 15, "Paris": 22, "Moscow": -2, "Sydney": 30}
  warm = {city: temp for city, temp in temperatures.items() if temp > 20}
  print(warm)  # {'Paris': 22, 'Sydney': 30}

  # Using enumerate with dict comprehension
  indexed = {i: val for i, val in enumerate(["a", "b", "c"])}
  print(indexed)  # {0: 'a', 1: 'b', 2: 'c'}

8. Merge & Update Operators (Python 3.9+)

The | operator merges two dicts into a new one. The |= operator updates a dict in place. Right-hand dict wins on duplicate keys:

defaults = {"theme": "light", "font_size": 12, "show_sidebar": True}
  user_prefs = {"theme": "dark", "font_size": 14}

  # Merge — create a new dict (right wins)
  config = defaults | user_prefs
  print(config)  # {'theme': 'dark', 'font_size': 14, 'show_sidebar': True}

  # The original dicts are unchanged
  print(defaults)  # {'theme': 'light', 'font_size': 12, 'show_sidebar': True}

  # Multiple merge — chain with |
  env_overrides = {"font_size": 16}
  final = defaults | user_prefs | env_overrides
  print(final)  # {'theme': 'dark', 'font_size': 16, 'show_sidebar': True}

  # In-place update (like .update() but more readable)
  settings = {"a": 1, "b": 2}
  settings |= {"b": 99, "c": 3}
  print(settings)  # {'a': 1, 'b': 99, 'c': 3}

  # Contrast with .update()
  settings2 = {"a": 1, "b": 2}
  settings2.update({"b": 99, "c": 3})
  print(settings2)  # {'a': 1, 'b': 99, 'c': 3} — same result

  # | creates a new dict; |= modifies in place
  original_id = id(settings2)
  settings2 |= {"d": 4}
  print(id(settings2) == original_id)  # True — in-place mutation

  # Merge with non-dict mappings also works
  print({"a": 1} | [("b", 2)])  # {'a': 1, 'b': 2}
💡 | vs update(): Use | when you want a new merged dict without modifying the originals. Use |= or .update() when you want to modify a dict in place. The | operator is more readable for merging and works great in functional-style code or one-liners.

9. Nested Dictionaries

Dicts can contain other dicts, creating hierarchical data structures:

# Nested dict — dict of dicts
  grades = {
      "Alice": {"Math": 95, "Science": 88, "English": 91},
      "Bob": {"Math": 87, "Science": 92, "English": 85},
      "Charlie": {"Math": 93, "Science": 90, "English": 89},
  }

  # Accessing nested values
  print(grades["Alice"]["Math"])      # 95

  # Adding a grade
  grades["Alice"]["History"] = 90
  print(grades["Alice"])              # {'Math': 95, 'Science': 88, 'English': 91, 'History': 90}

  # Iterating over nested dicts
  for student, subjects in grades.items():
      total = sum(subjects.values())
      count = len(subjects)
      print(f"{student}: {total/count:.1f} avg ({total} pts across {count} subjects)")

  # Deep nesting
  school = {
      "Math Dept": {
          "AL101": {"name": "Algebra I", "teacher": "Smith", "students": 30},
          "GE201": {"name": "Geometry", "teacher": "Jones", "students": 25},
      },
      "Science Dept": {
          "PH101": {"name": "Physics I", "teacher": "Brown", "students": 28},
      },
  }
  print(school["Math Dept"]["AL101"]["teacher"])  # Smith

  # Safe access with .get() on nested dicts
  course = school.get("History Dept", {})
  print(course.get("HI101", {}).get("name", "N/A"))  # N/A
⚠️ Deep .get() chaining: Chaining .get() on deeply nested dicts is verbose. For complex nested data, consider collections.defaultdict with nested patterns or library utilities.

10. __missing__ Hook & collections.defaultdict

Python's __missing__ hook is called when a key is not found via bracket access. The collections.defaultdict class implements this hook to provide automatic default values:

from collections import defaultdict

  # defaultdict(list) — auto-initialize missing keys with empty list
  fruits_by_color = defaultdict(list)

  fruits_by_color["red"].append("apple")
  fruits_by_color["red"].append("cherry")
  fruits_by_color["yellow"].append("banana")
  fruits_by_color["green"].append("apple")

  print(dict(fruits_by_color))
  # {'red': ['apple', 'cherry'], 'yellow': ['banana'], 'green': ['apple']}

  # defaultdict(int) — auto-initialize missing keys with 0
  word_counts = defaultdict(int)
  text = "the quick brown fox jumps over the lazy dog"
  for word in text.split():
      word_counts[word] += 1

  print(dict(word_counts))
  # {'the': 2, 'quick': 1, 'brown': 1, 'fox': 1, 'jumps': 1, 'over': 1, 'lazy': 1, 'dog': 1}

  # defaultdict(set) — auto-initialize with empty set (for unique grouping)
  categories = defaultdict(set)
  data = [("fruit", "apple"), ("fruit", "apple"), ("fruit", "banana"), ("color", "red")]
  for category, item in data:
      categories[category].add(item)

  print(dict(categories))  # {'fruit': {'banana', 'apple'}, 'color': {'red'}}

  # defaultdict with custom factory
  counter = defaultdict(lambda: 0)
  for ch in "hello world":
      counter[ch] += 1
  print(dict(counter))  # {'h': 1, 'e': 1, 'l': 3, 'o': 2, ' ': 1, 'w': 1, 'r': 1, 'd': 1}

  # Nested defaultdict — defaultdict of defaultdict
  nested = defaultdict(lambda: defaultdict(int))
  nested["Alice"]["Math"] += 95
  nested["Alice"]["Science"] += 88
  nested["Bob"]["Math"] += 92
  print(dict(nested))
  # {'Alice': defaultdict(, {'Math': 95, 'Science': 88}), ...}

  # Comparison with setdefault() — more verbose
  plain = {}
  for word in text.split():
      plain[word] = plain.get(word, 0) + 1
  print(plain == dict(word_counts))  # True — but defaultdict is cleaner

  # setdefault() for list grouping
  grouped = {}
  for color, fruit in [("red", "apple"), ("red", "cherry"), ("yellow", "banana")]:
      grouped.setdefault(color, []).append(fruit)
  print(grouped)  # {'red': ['apple', 'cherry'], 'yellow': ['banana']}

  # Caveat: defaultdict's default_factory applies to __getitem__ only
  # .get() still returns None for missing keys
  dd = defaultdict(int)
  dd["a"] += 1
  print(dd.get("b"))  # None — NOT 0!
💡 defaultdict vs setdefault(): defaultdict is cleaner when you want all missing keys to get the same default. Use setdefault() when only specific accesses should create defaults, or when you need different defaults per key.

11. collections.Counter

Counter is a specialized dict subclass for counting hashable objects. It's one of the most useful tools in the collections module:

from collections import Counter

  # Creating a Counter
  c = Counter("abracadabra")
  print(c)  # Counter({'a': 5, 'b': 2, 'r': 2, 'c': 1, 'd': 1})

  # Counter from a list
  colors = ["red", "blue", "red", "green", "blue", "red"]
  c2 = Counter(colors)
  print(c2)  # Counter({'red': 3, 'blue': 2, 'green': 1})

  # most_common(n) — the n most frequent elements
  print(c.most_common(3))  # [('a', 5), ('b', 2), ('r', 2)]

  # Accessing counts (returns 0 for missing — no KeyError!)
  print(c["z"])            # 0

  # elements() — iterator over elements repeating each as many times as its count
  print(sorted(c.elements()))
  # ['a', 'a', 'a', 'a', 'a', 'b', 'b', 'c', 'd', 'r', 'r']

  # total() — sum of all counts (Python 3.10+)
  print(c.total())         # 11

  # subtract() — decrement counts
  c2 = Counter("abracadabra")
  c2.subtract("alphabet")
  print(c2)  # Counter({'a': 3, 'r': 2, 'b': 1, 'c': 1, 'd': 1, 'l': -1, 'p': -1, 'h': -1, 'e': -1, 't': -1})

  # Counter arithmetic — returns new Counter with non-negative counts
  c1 = Counter(a=3, b=1)
  c2 = Counter(a=1, b=2, c=1)

  # Addition
  print(c1 + c2)  # Counter({'a': 4, 'b': 3, 'c': 1})

  # Subtraction (keeps only positive counts)
  print(c1 - c2)  # Counter({'a': 2})

  # Intersection (min)
  print(c1 & c2)  # Counter({'a': 1, 'b': 1})

  # Union (max)
  print(c1 | c2)  # Counter({'a': 3, 'b': 2, 'c': 1})

  # Unary plus/minus — remove non-positive counts
  c3 = Counter(a=3, b=-1, c=0)
  print(+c3)      # Counter({'a': 3}) — only positive counts
  print(-c3)      # Counter({'b': 1}) — only negative counts, made positive

  # Rich comparisons — subset/superset
  print(Counter(a=2, b=1) <= Counter(a=3, b=2, c=1))  # True

  # update() adds counts (unlike regular dict update which replaces)
  c4 = Counter(a=2, b=1)
  c4.update(Counter(a=1, b=2, c=3))
  print(c4)  # Counter({'a': 3, 'b': 3, 'c': 3})

  # Counter supports setdefault, pop, popitem, keys, values, items — like dict
💡 Counter vs defaultdict(int): Both can count, but Counter adds most_common(), elements(), total(), arithmetic operators, and rich comparisons. Use Counter when you need these features; use defaultdict(int) for simple counting in local scope.

12. collections.OrderedDict

Before Python 3.7 guaranteed insertion-order preservation in regular dicts, OrderedDict was the only ordered mapping. Today, regular dicts preserve order too, but OrderedDict still offers unique features:

from collections import OrderedDict

  # Creating an OrderedDict
  od = OrderedDict([("a", 1), ("b", 2), ("c", 3)])
  print(od)  # OrderedDict([('a', 1), ('b', 2), ('c', 3)])

  # move_to_end(key, last=True) — move key to end (or beginning if last=False)
  od.move_to_end("a")           # move 'a' to the end
  print(od)  # OrderedDict([('b', 2), ('c', 3), ('a', 1)])

  od.move_to_end("a", last=False)  # move 'a' to the beginning
  print(od)  # OrderedDict([('a', 1), ('b', 2), ('c', 3)])

  # popitem(last=True) — pop from end; popitem(last=False) — pop from beginning
  print(od.popitem())         # ('c', 3) — default last=True
  print(od.popitem(last=False))  # ('a', 1) — from the beginning

  # Order-sensitive equality
  od1 = OrderedDict([("a", 1), ("b", 2)])
  od2 = OrderedDict([("b", 2), ("a", 1)])
  print(od1 == od2)  # False — order matters for OrderedDict!
  print(dict(od1) == dict(od2))  # True — regular dict equality ignores order

  # Reinserting a key moves it to the end
  od3 = OrderedDict([("a", 1), ("b", 2), ("c", 3)])
  od3["a"] = 99
  print(od3)  # OrderedDict([('b', 2), ('c', 3), ('a', 99)])

  # When to use OrderedDict vs regular dict
  # 1. You need order-sensitive equality checks
  # 2. You need move_to_end() functionality
  # 3. You need popitem(last=False) — pop from beginning
  # 4. You want to support older Python versions (pre-3.7)

  # LRU cache pattern with move_to_end
  lru = OrderedDict()
  lru["a"] = 1
  lru["b"] = 2
  lru["c"] = 3

  # Simulating an access — move to end to mark as "recently used"
  def access(key):
      if key in lru:
          lru.move_to_end(key)
          return lru[key]
      return None

  print(access("a"))   # 1 — 'a' moves to end
  print(lru)           # OrderedDict([('b', 2), ('c', 3), ('a', 1)])
  print(access("b"))   # — 'b' is now at the beginning (least recently used before access)
💡 OrderedDict or regular dict? For most modern Python code (3.7+), use a regular dict — it preserves insertion order, has a faster C implementation, and is more idiomatic. Reach for OrderedDict only when you specifically need move_to_end(), order-sensitive equality, or pop from the beginning with popitem(last=False).
